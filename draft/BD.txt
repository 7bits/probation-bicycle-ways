CREATE DATABASE likeByke character set utf8 collate utf8_general_ci;
GRANT All ON likeByke.* TO 'sbogdanov'@'localhost' IDENTIFIED BY 'mysql';
CREATE TABLE Route(id int(20) not null, owner varchar(50) not null, primary key(id)) engine=InnoDB character set=utf8;        
create table Groups(id int(20) not null, `name` varchar(50) not null, `departament_id` int(20) not null, primary key(id), foreign key (departament_id) references Departaments(id)) engine=InnoDB character set=utf8;
create table Chairs(id int(20) not null, `name` varchar(50) not null, `departament_id` int(20) not null, primary key(id), foreign key (departament_id) references Departaments(id)) engine=InnoDB character set=utf8;
create table Students(id int(20) not null, `first_name` varchar(50) not null,
 `last_name` varchar(50) not null, `is_head` boolean not null, `group_id` int(20) not null,
  `chair_id` int(20), `departament_id` int(20) not null, primary key(id),
   foreign key (group_id) references Groups(id), foreign key (chair_id) references Chairs(id),
    foreign key (departament_id) references Departaments(id) ) engine=InnoDB character set=utf8;
ALTER TABLE Departaments ADD dean varchar(50);
